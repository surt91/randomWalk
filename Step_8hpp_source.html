166 template <>
167 inline std::vector<Step<int>> Step<int>::neighbors(bool diagonal) const
168 {
169  // A d dimensional hypercube has 2*d direct neighbors, in positive
170  // and negative direction for every dimension.
171  int num_neighbors = 2*m_d;
172  if(diagonal)
173  {
174  if(m_d == 2)
175  num_neighbors += 4;
176  if(m_d == 3)
177  num_neighbors += 8 + 12;
178  }
179 
180  std::vector<Step<int>> ret(num_neighbors, *this);
181  for(int i=0; i<m_d; ++i)
182  {
183  ret[2*i][i] += 1;
184  ret[2*i+1][i] -= 1;
185  }
186 
187  if(diagonal)
188  {
189  // build a lookup table ... actually, I am not sure if it gets faster at all
190  #if D_MAX == 0
191  std::vector<int> pow2(m_d);
192  pow2[0] = 1;
193  for(int i = 1; i<=D_MAX; ++i)
194  pow2[i] = pow2[i-1] * 2;
195  #else
196  int pow2[D_MAX+1];
197  pow2[0] = 1;
198  for(int i = 1; i<=D_MAX; ++i)
199  pow2[i] = pow2[i-1] * 2;
200  #endif
201 
202  if(m_d > 3)
203  throw std::invalid_argument("not implemented for d > 3");
204 
205  Step<int> diff(m_d);
206  // A d dimensional hypercube has 2^d diagonal elements
207  // we can enumerate them with bitfiddling, i.e., to
208  // generate all 2^d possibilities, we take all numbers
209  // up to 2^d-1 and interpret their j-th bit as a displacement
210  // in negative direction for 0 and positive for 1.
211  // The sum of the center and this displacement generates
212  // all cornerstones.
213  for(int i=0; i<pow2[m_d]; ++i)
214  {
215  for(int j=0; j<m_d; ++j)
216  diff[j] = (i & (1 << j)) ? 1 : -1;
217  ret[2*m_d + i] += diff;
218  }
219 
220  // in d = 3 there are also the sides
221  // for each direction, leave it at zero and generate all combinations
222  // of the remaining two directions
223  if(m_d == 3)
224  for(int i=0; i<m_d; ++i)
225  for(int j=0; j<pow2[m_d-1]; ++j)
226  {
227  diff[i] = 0;
228  for(int k=0; k<m_d-1; ++k)
229  {
230  if(i <= k)
231  diff[k+1] = (j & (1 << k)) ? 1 : -1;
232  else
233  diff[k] = (j & (1 << k)) ? 1 : -1;
234  }
235  ret[2*m_d + pow2[m_d] + i*pow2[m_d-1] + j] += diff;
236  }
237 
238  // I can not imagine d > 3 easily. There should be one category more
239  // where two coordinates are the same and two change.
240  // I will implement it as soon as I need it.
241  }
242 
243  return ret;
244 }
