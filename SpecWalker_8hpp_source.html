504 template <class T>
505 double SpecWalker<T>::oblateness() const
506 {
507  double maxLongAxis = 0.;
508  int maxI = 0;
509  int maxJ = 0;
510  int n_hullpoints = hullPoints().size();
511 
512  // first: search the two points with largest distance
513  for(int i=0; i<n_hullpoints; ++i)
514  for(int j=0; j<i; ++j)
515  {
516  double diameter = (hullPoints()[i] - hullPoints()[j]).length();
517  if(diameter > maxLongAxis)
518  {
519  maxLongAxis = diameter;
520  maxI = i;
521  maxJ = j;
522  }
523  }
524 
525  double maxShortAxis = 0.;
526  double maxCross = 0.;
527  // second search point with largest distance to the axis between i and j
528  // that is the point p which (in d = 2)
529  // maximizes |cross2d(i, p, j)|
530  for(int k=0; k<n_hullpoints; ++k)
531  {
532  double cross = std::abs(cross2d_z(hullPoints()[maxI], hullPoints()[k], hullPoints()[maxJ]));
533  if(cross > maxCross)
534  {
535  maxShortAxis = cross / hullPoints()[maxI].dist(hullPoints()[maxJ]);
536  maxCross = cross;
537  }
538  }
539 
540  // divide by 2 maxShortAxis, to give 1 for a sphere
541  // will result in unintuitive values for asymmetric shapes
542  return maxLongAxis/maxShortAxis/2.;
543 }

603 template <>
604 inline int SpecWalker<int>::enclosedSites() const
605 {
606  // derive a bounding box from the convex hull,
607  // extend it by 1 in every direction
608  // perform a DFS (visited sites can not be entered)
609  // every not-encountered site is enclosed by the walk
610  std::unordered_set<Step<int>> distinctPoints;
611  std::unordered_set<Step<int>> encountered_sites;
612  std::vector<Step<int>> stack;
613 
614 
615  std::vector<int> max(d, 0), min(d, 0);
616  for(auto &p : points())
617  {
618  for(int i=0; i<d; ++i)
619  {
620  if(p[i] > max[i])
621  max[i] = p[i];
622  if(p[i] < min[i])
623  min[i] = p[i];
624  }
625  distinctPoints.insert(p);
626  }
627 
628  for(int i=0; i<d; ++i)
629  {
630  ++max[i];
631  --min[i];
632  }
633 
634  int size = 1;
635  for(int i=0; i<d; ++i)
636  {
637  size *= max[i] - min[i] + 1;
638  }
639 
640  // start in a corner
641  Step<int> start(max);
642  stack.push_back(start);
643  encountered_sites.insert(start);
644  while(!stack.empty())
645  {
646  Step<int> current = stack.back();
647  stack.pop_back();
648 
649  for(auto &n : current.neighbors())
650  {
651  bool inside_bounds = true;
652  for(int i=0; i<d; ++i)
653  if(n[i] > max[i] || n[i] < min[i])
654  inside_bounds = false;
655 
656  if(inside_bounds && !encountered_sites.count(n) && !distinctPoints.count(n))
657  {
658  stack.push_back(n);
659  encountered_sites.insert(n);
660  }
661  }
662  }
663 
664  return size - encountered_sites.size();
665 }

768 int SpecWalker<T>::passage(int t1, int axis) const
769 {
770  if(t1 >= numSteps)
771  return -1;
772  int startSign = sign(points()[t1].x(axis));
773  // if we are on zero, search the next non-null coordinate
774  while(startSign == 0)
775  {
776  if(t1 >= numSteps)

780  for(int i=t1+1; i<numSteps; ++i)
781  {

784  {
785  return i;
786  }
787  }
788  return -1;
