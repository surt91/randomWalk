824  {
825  LOG(LOG_ERROR) << "Jarvis March is only implemented in d=2, the data is d = " << d;
826  throw std::invalid_argument("Jarvis March is only implemented in d=2");
827  }
828 
829  // Step < sorts by x value
830  Step<T> p, p1;
831 
832  std::unordered_set<Step<T>> candidate_points;
833  if(algorithm != CH_JARVIS_AKL)
834  {
835  candidate_points = std::unordered_set<Step<T>>(interiorPoints->begin(), interiorPoints->end());
836  p1 = std::min(*interiorPoints);
837  }
838  else
839  {
840  for(int i=0; i<n; ++i)
841  candidate_points.insert(*pointSelection[i]);
842  p1 = std::min(*interiorPoints);
843  }
844 
845  candidate_points.erase(p1);
846  hullPoints_.push_back(p1);
847 
848  int hull_idx = 0;
849  do
850  {
851  // Search for a point 'p' such that orientation(hull.last, i, p) is
852  // counterclockwise for all points 'i'
853  p = p1;
854 
855  for(auto it = candidate_points.begin(); it != candidate_points.end(); ++it)
856  {
857  T orientation = cross2d_z(hullPoints_[hull_idx], *it, p);
858  if(orientation > 0)
859  p = *it;
860  else if(orientation == 0) // colinear
861  {
862  // take the one furthest away
863  if((p-hullPoints_[hull_idx]).length() < (*it-hullPoints_[hull_idx]).length())
864  p = *it;
865  // and delete all other
866  }
867  }
868 
869  hull_idx++;
870  hullPoints_.push_back(p);
871 
872  // found, then elimiminate all points inside the triangle (hull[0], hull[i-1], hull[i])
873  candidate_points.erase(p);
874 
875  for(auto it = candidate_points.begin(); it != candidate_points.end(); )
876  {
877  if(pointInTriangle(hullPoints_[0],
878  hullPoints_[hull_idx-1],
879  hullPoints_[hull_idx],
880  *it))
881  it = candidate_points.erase(it);
882  else
883  ++it;
884  }
885  } while(p!=p1); // if we reach the first, we have finished
886  // mind that first and last entry of hullPoints_ are the same
887 }
