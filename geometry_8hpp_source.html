221 template <class T>
222 T side(const Step<T>& p1,
223  const Step<T>& p2,
224  const Step<T>& p3,
225  const Step<T>& p)
226 {
227  // calculate dot product of
228  // 1. vector from p to some point on the face
229  // 2. normal vector (obtained by cross product)
230  // if they have a parallel component -> behind -> dot > 0
231  // if they have a anitparallel component -> infront -> dot < 0
232 
233  //~ return dot(p1-p, cross(p2-p1, p3-p1));
234 
235  // formulate it directly by hand, to avoid temporaries
236  // totally premature optimization, but speeds it up somewhat
237  auto c1x = p2.x() - p1.x();
238  auto c2x = p3.x() - p1.x();
239  auto c1y = p2.y() - p1.y();
240  auto c2y = p3.y() - p1.y();
241  auto c1z = p2.z() - p1.z();
242  auto c2z = p3.z() - p1.z();
243  return (p1.x()-p.x()) * (c1y*c2z - c1z*c2y)
244  + (p1.y()-p.y()) * (c1z*c2x - c1x*c2z)
245  + (p1.z()-p.z()) * (c1x*c2y - c1y*c2x);
246 }
